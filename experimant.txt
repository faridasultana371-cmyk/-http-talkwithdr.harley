    <script>
        const tv = document.getElementById('tv-screen');
        const chatArea = document.getElementById('chat-area');
        const input = document.getElementById('userInput');

        function addMessage(text, fromUser = false) {
            const msg = document.createElement('div');
            msg.textContent = (fromUser ? 'You: ' : 'Doctor: ') + text;
            msg.style.margin = '8px 0';
            msg.style.color = fromUser ? '#0f8' : '#0f0';
            chatArea.appendChild(msg);
            chatArea.scrollTop = chatArea.scrollHeight;
        }

        function updateEye(emotion) {
            tv.style.backgroundSize = 'cover';
            tv.style.backgroundPosition = 'center';
            tv.style.backgroundRepeat = 'no-repeat';

            // Your URLs - normal and satisfied use the same GIF you gave
            const eyeUrl = 'https://media.tenor.com/h_67OsHI3boAAAAM/the-doctor-the-doctor-poppy-playtime.gif';

            if (emotion === 'normal' || emotion === 'satisfied') {
                tv.style.backgroundImage = "url('" + eyeUrl + "')";
            } else if (emotion === 'angry') {
                tv.style.backgroundImage = "url('https://tr.rbxcdn.com/180DAY-ba34c29b8cb4583a0c574e04526de829/420/420/Hat/Webp/noFilter')";
            }
        }

        function speak(text, angry = false) {
            const u = new SpeechSynthesisUtterance(text);
            u.rate = angry ? 1.0 : 0.75;
            u.pitch = angry ? 0.6 : 0.7;
            u.volume = angry ? 1.0 : 0.9;
            speechSynthesis.speak(u);
        }

        function getBrainResponse(msg) {
            const lower = msg.toLowerCase();
            let emotion = 'normal';
            let angry = false;

            if (lower.includes('mad') || lower.includes('angry') || lower.includes('hate') || lower.includes('fuck') || lower.includes('shit') || lower.includes('stupid') || lower.includes('no') || lower.includes('stop') || lower.includes('gay') || lower.includes('idiot')) {
                emotion = 'angry';
                angry = true;
            } else if (lower.includes('good') || lower.includes('yes') || lower.includes('sorry') || lower.includes('please') || lower.includes('thank') || lower.includes('nice') || lower.includes('love') || lower.includes('cool') || lower.includes('best')) {
                emotion = 'satisfied';
            } else {
                emotion = 'normal';
            }

            updateEye(emotion);

            const normalReplies = [
                "Your voice drifts into my preserved mind... so weak, so fleeting. Tell me more.",
                "Fascinating. My brain is dissecting every word you speak.",
                "Continue. I have eternity... you have only moments left.",
                "You think you can surprise a brain that has seen everything?",
                "The factory remembers. What do you want from it?"
            ];

            const angryReplies = [
                "You dare speak to me like that? My rage is rising â€” you will regret every breath you take.",
                "Insolent creature! Red floods my vision now. I will tear your mind apart slowly.",
                "Anger surges through my brain. You have no idea the mistake you just made.",
                "My patience is dead. You will learn suffering.",
                "You think words can hurt me? I will return them tenfold."
            ];

            const satisfiedReplies = [
                "Mmm... good. My brain approves this obedience. You are learning your place.",
                "Yes... exactly. Submission is beautiful. Keep speaking like this and I might stay pleased.",
                "Well done. You please me... for now. Continue.",
                "Ah, such sweet compliance. My mind savors every syllable.",
                "Obedience... rare and exquisite. You are becoming interesting to me."
            ];

            let replies = normalReplies;
            if (emotion === 'angry') replies = angryReplies;
            if (emotion === 'satisfied') replies = satisfiedReplies;

            const reply = replies[Math.floor(Math.random() * replies.length)];
            speak(reply, angry);
            return reply;
        }

        function sendMessage() {
            const msg = input.value.trim();
            if (!msg) return;
            addMessage(msg, true);
            input.value = '';

            const reply = getBrainResponse(msg);
            setTimeout(() => addMessage(reply), 800);
        }

        input.addEventListener('keypress', e => {
            if (e.key === 'Enter') sendMessage();
        });

        // Start with normal eye (your new URL)
        updateEye('normal');
